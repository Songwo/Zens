    @Override
    public LoginResponse login(LoginReq req, String deviceId, String clientIp) {
        String loginType = req.getLoginType();
        if (!StringUtils.hasText(loginType)) {
            loginType = StringUtils.hasText(req.getCode()) ? "otp" : "password";
        }
        User user = "otp".equals(loginType)
                ? loginByOtp(req)
                : loginByPassword(req);
        return finalizeLogin(
                user,
                req.isRememberMe(),
                normalizeDeviceId(deviceId),
                null,
                clientIp,
                req.getTwoFactorCode());
    }

    private LoginResponse finalizeLogin(User user, boolean rememberMe,
            String deviceId, String existsSessionId, String clientIp, String twoFactorCode) {
        ensureUserCanLogin(user);
        if (isTwoFactorEnabled(user) && !StringUtils.hasText(twoFactorCode)) {
            return buildTwoFactorChallenge(user, rememberMe, deviceId, clientIp);
        }
        if (isTwoFactorEnabled(user) && !TotpUtil.verifyCode(user.getTwoFactorSecret(), twoFactorCode)) {
            throw new LoginException("二步验证码错误");
        }
        String sessionId = UUID.randomUUID().toString();
        return generateTokenResponse(user, rememberMe, deviceId, sessionId, clientIp);
    }

    private LoginResponse generateTokenResponse(User user, boolean rememberMe,
            String deviceId, String sessionId, String clientIp) {
        Map<String, Object> baseClaims = jwtUtil.buildClaims(
                user.getUsername(), List.of(user.getRole()), user.getAvatar());
        baseClaims.put("sid", sessionId);
        baseClaims.put("did", deviceId);
        baseClaims.put("remember", rememberMe);

        Map<String, Object> accessClaims = new HashMap<>(baseClaims);
        accessClaims.put("typ", "access");
        Map<String, Object> refreshClaims = new HashMap<>(baseClaims);
        refreshClaims.put("typ", "refresh");

        String accessToken = jwtUtil.generateAccessToken(user.getId(), accessClaims, rememberMe);
        String refreshToken = jwtUtil.generateRefreshToken(user.getId(), refreshClaims, rememberMe);
        stringRedisTemplate.opsForValue().set(buildAccessKey(user.getId(), sessionId), accessToken);
        stringRedisTemplate.opsForValue().set(buildRefreshKey(user.getId(), sessionId), hashToken(refreshToken));
        stringRedisTemplate.opsForValue().set(buildDeviceKey(user.getId(), sessionId), deviceId);

        LoginResponse response = new LoginResponse();
        response.setAccessToken(accessToken);
        response.setRefreshToken(refreshToken);
        response.setTwoFactorRequired(false);
        return response;
    }

    private User loginByPassword(LoginReq req) {
        String account = req.getAccount();
        User user = account.contains("@")
                ? userService.lambdaQuery().eq(User::getEmail, account).one()
                : userService.lambdaQuery().eq(User::getUsername, account).one();
        if (user == null || !passwordEncoder.matches(req.getPassword(), user.getPassword())) {
            throw new LoginException("账号或密码错误");
        }
        return user;
    }

    private User loginByOtp(LoginReq req) {
        if (!verificationCodeService.verifyCode(req.getEmail(), req.getCode())) {
            throw new LoginException("验证码错误或已失效");
        }
        User user = userService.lambdaQuery()
                .eq(User::getEmail, req.getEmail())
                .one();
        if (user == null) {
            throw new LoginException("该邮箱尚未注册，请先注册");
        }
        return user;
    }
